model class

function selectUser // После логина вернуть все данные по юзеру
function selectTasks // Достать юзера задачи из бд, вызывается в selectUser
function login // Залогиниться/зарегистрировать нового пользователя с обработкой ошибок, по полученным данным из getLogin

controllers class

function getLoginForm // Если пользователь не залогинен - возвращает true
function getUser // Вызывается, если пользователь залогинился, если getLoginForm - false, возвращает имя пользователя, 
                    возможность добавлять задачи и список его задач с возможностью удаления/смены статуса
function getTasks // Возвращает задачи, вызывается в getUser
function getProfile // Возвращает имя пользователя, кнопку выхода, вызывается в getUser
function getLogin // Возвращает данные, которые пользователь ввёл в форму для логина, вызывается в login

view class

function drawPage // Выводит всю страницу
function drawLoginForm // Если сработала getLoginForm - true, выводит html для логина, вызывается в drawPage
function drawProfile // Выводит на страницу имя пользователя и кнопку выхода, вызывается в drawPage
function drawTasks // Выводит вёрстку задач, с возможностью удаления/смены статуса, вызывается в drawPage
